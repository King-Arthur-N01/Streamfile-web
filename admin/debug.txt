<?php
function getDirectoryFiles($dir) {
    $files = array();
    if (is_dir($dir)) {
        if ($handle = opendir($dir)) {
            while (($file = readdir($handle)) !== false) {
                $files[] = $file;
            }
            closedir($handle);
        }
    }
    return $files;
}

    $files = getDirectoryFiles($dir);

    }
}
